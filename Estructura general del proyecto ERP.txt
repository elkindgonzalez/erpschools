clean-architecture-project/
├── client/      ← Frontend con React (Vite)
│   ├── src/
│   │   ├── api/           ← peticiones al backend (axios o fetch)
│   │   ├── components/    ← componentes reutilizables
│   │   ├── hooks/         ← lógica compartida (useAuth, etc.)
│   │   ├── pages/         ← vistas
│   │   ├── routes/        ← enrutador React Router
│   │   ├── store/         ← estado global
│   │   └── utils/         ← funciones utilitarias
│   ├── public/
│   ├── package.json
│   └── vite.config.ts
│
└── server/      ← Backend con Node.js + Express
    ├── src/
    │   ├── application/   ← casos de uso (use cases)
    │   ├── domain/        ← entidades y reglas del negocio puras
    │   ├── infrastructure/← DB, repositorios, adaptadores externos
    │   ├── interfaces/    ← controladores, rutas, middlewares
    │   ├── config/        ← env, logger, seguridad
    │   ├── shared/        ← utilidades compartidas
    │   ├── tests/         ← pruebas
    │   ├── app.ts         ← configuración Express
    │   └── index.ts       ← arranque del servidor
    ├── package.json
    ├── tsconfig.json
    └── .env
